```Kotlin
/**
 * 4-Operators
 * + Toplama
 * - Çıkarma
 * * Çarpma
 * / Bölme
 * % Mod alma
 */
var num1 = 8
var num2 = 34

println(num1 + num2) //42
println(num2 - num1) //26
println(num1 * num2) //272
println(num2 / num1) //4
println(num2 % num1) //2

//Aşağıdaki kodun çıktısı nedir?
var num1 = 3
var num2 = 6
var x = num2 / num1
println(num2 / x)
//Cevap: 3

//Atama operatörü "=" değişkenlere değer atarken kullanılır.
//Kotlin aritmetik operatörleri de destekler.
//Örn.: a += b ifadesi ile a = a + b ifadeleri eşittir.
var num = 4
num *= 5 //num = 20

//iki değişkenin toplamı:
var a = 9
var b = 15
var res = a + b
println(res)

//Karşılaştırma operatörleri 
//>, <, >=, <=, ==, !=

println(age >= 16) //true
println(a==b) //a ve b eşit midir? t/f
```
